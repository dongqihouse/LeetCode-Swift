class Solution {
    func nextGreaterElement(_ nums1: [Int], _ nums2: [Int]) -> [Int] {
        
        var nums2Map = [Int: Int]()
        for i in 0..<nums2.count {
            for j in i..<nums2.count {
                if nums2[j] > nums2[i] {
                    nums2Map[nums2[i]] = nums2[j]
                    break;
                }
            }
        }
        var res = [Int]()
        for i in 0..<nums1.count {
            res.append(nums2Map[nums1[i], default: -1])
        }

        return res
    }
}
